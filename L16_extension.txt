1)
	Syntax for zip extesion    :
							
		zip filename.zip filename

		EX:

			zip dummy.zip dummy

						

	Syntax for unzip extesion    :


		unzip filename.zip 

		EX:

			unzip dummy.zip



2)

	Syntax for tar extesion    :

							
		tar cvf filename.tar filename

		EX: 

			tar cvf dummy1.tar dummy1



	Syntax for untar extension :

		tar xvf filename.tar

		EX : 

			tar xvf dummy1.tar





3)

	Syntax for gz extesion 	:

		gzip filename.txt

		EX :

			gzip dummy2.txt

			Output : dummy2.txt.gz


	Syntax for gunzip extension :

		gunzip filename.txt.gz

		EX : 

			gunzip dummy2.txt.gz

			Output : dummy2.txt
